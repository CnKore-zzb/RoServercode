--skill技能图标全读取处
local Skills = {}
local tmp = {}

UIAtlasConfig.IconAtlas.Skill = Skills

for k,v in pairs(UIAtlasConfig.IconAtlas) do
	if(string.find(k,"SkillProfess")~=nil) then
		for i=1,#v do
			if(tmp[v[i]]==nil) then
				tmp[v[i]] = 1
				Skills[#Skills+1] = v[i]
			end
		end
	end
end
